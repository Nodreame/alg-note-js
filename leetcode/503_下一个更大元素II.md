### 方法 1 单调栈

- 复杂度：
    - 时间 O(N)
    - 空间 O(N)

``` js
var nextGreaterElements = function(nums) {
    if (!nums || nums.length === 0) return []
    let len = nums.length
    const res = Array.from({ length: len }, ()=>-1)
    const stack = []
    for (let i=0; i<len*2; i++) {
        let idx = i % len
        while (stack.length !== 0 && nums[idx]>nums[stack[stack.length-1]]) {
            res[stack.pop()] = nums[idx]
        }
        stack.push(idx)
    }
    return res
};
```
